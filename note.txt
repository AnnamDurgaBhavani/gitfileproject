public class AddThreeNumbers {

    public static void main(String[] args) {
        // Declare three variables to store the numbers
        int num1 = 10;  // You can change this value to any integer
        int num2 = 20;  // You can change this value to any integer
        int num3 = 30;  // You can change this value to any integer

        // Calculate the sum of the three numbers
        int sum = num1 + num2 + num3;

        // Print the result
        System.out.println("The sum of " + num1 + ", " + num2 + ", and " + num3 + " is: " + sum);
    }
}
